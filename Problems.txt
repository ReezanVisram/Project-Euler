    Private Function isPrime(ByVal x As Long) As Boolean

        'funtion to determine if the number is a prime number

        If x = 1 Then
            isPrime = True
            Exit Function
        End If
        For i = 2 To x - 1
            If x Mod i = 0 Then
                isPrime = False
                Exit Function
            End If
        Next

        isPrime = True

    End Function
